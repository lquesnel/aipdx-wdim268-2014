CORGI & BOXER

A simple JavaScript Slideshow displaying some images, with a 'back' button, and a 'next' button.

Function

MySlides=new Array('corgi1.jpg','corgi2.jpg','boxer1.jpg','boxer2.jpg')
Slide=0
function ShowSlides(SlideNumber) {
{ Slide=Slide+SlideNumber
if (Slide>MySlides.length-1) { Slide=0 }
if (Slide < 0) { Slide=MySlides.length - 1 }
document.DisplaySlide.src=MySlides[Slide]
}
}
